Sub GravarOrcamento()

    Dim wb As Workbook
    Dim nomeCliente As String
    Dim numeroOrcamento As String
    Dim caminho As Variant
    Dim nomeArquivo As String
    Dim resposta As VbMsgBoxResult

    Set wb = ThisWorkbook

    ' Gera número do orçamento no formato hhmmddmmyyyy
    numeroOrcamento = Format(Now, "hhmmddmmyyyy")
    
    ' Escreve na célula D5 da aba CAPA
    With wb.Sheets("CAPA")
        .Range("D5").Value = numeroOrcamento
        nomeCliente = Trim(.Range("D6").Value)
    End With
    
    ' Garante que tenha nome de cliente
    If nomeCliente = "" Then
        MsgBox "O nome do cliente (célula D6 na aba CAPA) está vazio.", vbExclamation
        Exit Sub
    End If
    
    ' Abre janela para escolher onde salvar a cópia
    caminho = Application.GetSaveAsFilename( _
        InitialFileName:=nomeCliente & " - Orçamento " & numeroOrcamento & ".xlsm", _
        FileFilter:="Planilha habilitada para macro (*.xlsm), *.xlsm", _
        Title:="Salvar cópia do orçamento")

    If caminho = False Then
        MsgBox "Operação cancelada.", vbInformation
        Exit Sub
    End If

    ' Salva a cópia
    wb.SaveCopyAs caminho

    MsgBox "Orçamento gravado com sucesso!" & vbCrLf & _
           "Número do orçamento: " & numeroOrcamento, vbInformation

    ' -------------------------------
    ' NOVO: Pergunta se deseja imprimir
    ' -------------------------------
    resposta = MsgBox("Deseja imprimir o layout em PDF?", vbYesNo + vbQuestion, "Imprimir PDF")

    If resposta = vbYes Then
        Call ImprimirDuasAbasPDF
    Else
        MsgBox "Orçamento finalizado sem impressão.", vbInformation
    End If

End Sub

Public Sub CriarBlocoEstruturaPreMoldada()
    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim linha As ListRow
    Dim etapa As String, descricao As String
    Dim quant As Variant, unidade As String
    Dim produtos(1 To 3) As String, quantidades(1 To 3) As Variant, unidades(1 To 3) As String
    Dim i As Integer, linhaDestino As Long
    Dim textoFinal As String
    Dim wsDestino As Worksheet
    Dim colDesc As Long, colLink As Long
    Dim linhaCadastro As Range
    Dim linkImagem As String
    Dim inicioTextoExtra As Long

    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tbl = wsLinear.ListObjects("Tabela24")

    linhaDestino = proximaLinha

   ' Buscar até 3 produtos da etapa "Tipo de construção"
i = 1
For Each linha In tbl.ListRows
    etapa = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
    
    If LCase(etapa) Like "tipo de construção*" Then
        descricao = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value))
        quant = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unidade = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value))
        
        If descricao <> "" And Val(quant) > 0 And i <= 3 Then
            ' Ignorar produtos que comecem com "sem" ou "serviço"
            If Not (LCase(Left(descricao, 3)) = "sem" Or LCase(Left(descricao, 7)) = "serviço") Then
                produtos(i) = descricao
                quantidades(i) = quant
                unidades(i) = unidade
                i = i + 1
            End If
        End If
    End If
Next linha


    ' Se encontrou algum produto, escreve o bloco
    If produtos(1) <> "" Then
        With wsImpressao.Cells(linhaDestino, 1)
            .Value = "TIPO DE CONSTRUÇÃO"
            .Font.Bold = True
            .Font.Size = 11
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With

        ' Montar a descrição
        For i = 1 To 3
            If produtos(i) <> "" Then
                textoFinal = textoFinal & produtos(i) & " – " & quantidades(i) & " " & unidades(i) & vbNewLine
            End If
        Next i

        ' Adiciona o texto institucional final
        textoFinal = textoFinal & vbNewLine & _
            "Na SLS, utilizamos um sistema exclusivo com placas de concreto pré-moldadas, que garantem montagem rápida, precisa e com acabamento superior." & vbNewLine & _
            "Nosso concreto é estanque, desenvolvido em laboratório com alta resistência e formulação que evita fissuras." & vbNewLine & _
            "A estrutura é leve, robusta e facilita o transporte e a instalação, acelerando a obra e reduzindo sujeira e desperdício." & vbNewLine & _
            "Com mais de 200 piscinas entregues, oferecemos engenharia confiável, acabamento de alto padrão e durabilidade comprovada."

        ' Escrever na coluna C e aplicar destaque no texto institucional
        With wsImpressao.Cells(linhaDestino, 3)
            .Value = textoFinal
            .Font.Name = "Calibri"
            .Font.Size = 9
            .Font.Bold = False
            .WrapText = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop

            ' Estilo personalizado no trecho institucional
            inicioTextoExtra = InStr(textoFinal, "Na SLS, utilizamos")
            If inicioTextoExtra > 0 Then
                With .Characters(Start:=inicioTextoExtra, Length:=Len(textoFinal) - inicioTextoExtra + 1).Font
                    .Color = RGB(0, 112, 192) ' Azul claro
                    .Name = "Segoe UI"
                    .Size = 7
                    .Bold = False
                End With
            End If
        End With

        ' Buscar imagem com prioridade: produto 1, 2, 3
        On Error Resume Next
        colDesc = wsDados.ListObjects("Cadastro").ListColumns("Descrição").Index
        colLink = wsDados.ListObjects("Cadastro").ListColumns("Link da Imagem").Index
        On Error GoTo 0

        linkImagem = ""
        For i = 1 To 3
            If produtos(i) <> "" Then
                For Each linhaCadastro In wsDados.ListObjects("Cadastro").DataBodyRange.Rows
                    If Trim(UCase(linhaCadastro.Cells(1, colDesc).Value)) = Trim(UCase(produtos(i))) Then
                        linkImagem = linhaCadastro.Cells(1, colLink).Value
                        Exit For
                    End If
                Next linhaCadastro
            End If
            If linkImagem <> "" Then Exit For
        Next i

        ' Inserir imagem ou mensagem
        With wsImpressao.Cells(linhaDestino, 5)
            If linkImagem <> "" Then
                .FormulaLocal = "=IMAGEM(" & Chr(34) & linkImagem & Chr(34) & ")"
            Else
                .Value = "Imagem não encontrada"
            End If
        End With

        ' Formatação da linha
        With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
            .Interior.Color = RGB(242, 242, 242)
            With .Borders(xlEdgeBottom)
                .LineStyle = xlContinuous
                .Color = RGB(180, 180, 180)
                .Weight = xlThin
            End With
        End With

        wsImpressao.Rows(linhaDestino).RowHeight = 140
        proximaLinha = linhaDestino + 1
    End If
End Sub

' Atualiza todas as tabelas da aba Imp.Tabela
Sub AtualizarImpTabelaCompleto()

    Dim Con As WorkbookConnection
    Dim ws As Worksheet
    Dim pt As PivotTable
    Dim shp As Shape
    Dim ultimaCelula As Range
    Dim areaImpressao As String
    Dim numeroOrcamento As String
    Dim wsImp As Worksheet

    On Error GoTo TrataErro
    Application.ScreenUpdating = False
    Application.EnableEvents = False
      
    ' Desprotege aba Imp.Tabela para permitir atualizações
    Set wsImp = ThisWorkbook.Sheets("Imp.Tabela")
    wsImp.Unprotect Password:="*Ju290480"
    
    ' Aguarda 2 segundos depois de desproteger
    Application.Wait (Now + TimeValue("0:00:01"))

        
    ' ---------- PASSO 1: Atualizar TODAS AS CONEXÕES ----------
    ThisWorkbook.RefreshAll

    ' ---------- PASSO 2: Atualizar TODAS AS PIVOTTABLES DA ABA Imp.Tabela ----------
    For Each pt In wsImp.PivotTables
        pt.RefreshTable
    Next pt

    ' Vai para a aba Imp.Tabela
    wsImp.Activate

    ' -------- Executa os blocos de impressão --------
    Call MontarLayoutOrcamento
    Call InserirQuebraPaginaCondicoes

    ' -------- Ajustes finais para impressão da aba Imp.Tabela --------
    With wsImp
        ' Limpa área de impressão antiga
        .PageSetup.PrintArea = ""

        ' Garante que não existam colunas ou linhas ocultas
        .Cells.EntireColumn.Hidden = False
        .Cells.EntireRow.Hidden = False

        ' Remove shapes abaixo da linha 1 (mantém botões da linha 1)
        For Each shp In .Shapes
            If shp.TopLeftCell.Row > 1 Then
                shp.Delete
            End If
        Next shp

        ' Identifica última célula usada
        Set ultimaCelula = .Cells.SpecialCells(xlCellTypeLastCell)

        ' Define nova área de impressão
        areaImpressao = .Range("A1", ultimaCelula).Address
        .PageSetup.PrintArea = areaImpressao

        ' Define orientação e margens
        .PageSetup.Orientation = xlPortrait
        .PageSetup.TopMargin = Application.CentimetersToPoints(2.5)
        .PageSetup.BottomMargin = Application.CentimetersToPoints(1)
        .PageSetup.LeftMargin = Application.CentimetersToPoints(0.6)
        .PageSetup.RightMargin = Application.CentimetersToPoints(0.6)

        ' Ajusta para caber em 1 página de largura (altura livre)
        With .PageSetup
            .Zoom = False
            .FitToPagesWide = 1
            .FitToPagesTall = False
        End With
    End With
       
    Application.EnableEvents = True
    Application.ScreenUpdating = True

    MsgBox "Consultas, Power Query e Tabelas Dinâmicas foram atualizadas com sucesso!", vbInformation
        
    
   ' Chama a Sub que protege a aba depois de tudo
Application.OnTime Now + TimeValue("00:00:07"), "ProtegerAbaImpTabela"
        
        Exit Sub
        
TrataErro:
    MsgBox "Erro ao atualizar as tabelas: " & Err.Description, vbCritical
    Application.EnableEvents = True
    Application.ScreenUpdating = True

End Sub

Sub InserirQuebraPaginaCondicoes()
    '
    ' Versao inteligente:
    ' - Remove todas as quebras manuais existentes.
    ' - Identifica a linha "CONDICOES GERAIS DA PROPOSTA" e a ultima linha
    '   de conteudo (fim do bloco CONDICOES + TabelaParcelas).
    ' - Forca o Excel a recalcular as quebras automaticas (via view switch).
    ' - Se houver uma quebra automatica DENTRO da secao CONDICOES (entre a
    '   linha dela e o ultimo conteudo) -> a secao seria rachada -> insere
    '   uma quebra manual ANTES de CONDICOES para joga-la na proxima pagina.
    ' - Se ja houver quebra automatica proxima ANTES de CONDICOES -> nao faz
    '   nada (o Excel ja vai paginar corretamente).
    ' - Se tudo couber na pagina atual sem rachar -> nao insere nada.
    '
    Dim wsImp As Worksheet
    Dim wnd As Window
    Dim r As Long, linhaCondicoes As Long, linhaFimSecao As Long
    Dim hpb As HPageBreak
    Dim breakNaSecao As Boolean
    Dim breakJustoAcima As Boolean
    Dim originalView As Long

    Set wsImp = ThisWorkbook.Sheets("Imp.Tabela")

    ' Limpa breaks manuais
    On Error Resume Next
    wsImp.Cells.PageBreak = xlPageBreakNone
    On Error GoTo 0

    ' Acha linha CONDICOES
    linhaCondicoes = 0
    Dim ultimaLinha As Long
    ultimaLinha = wsImp.UsedRange.Rows.Count + wsImp.UsedRange.Row - 1
    For r = 1 To ultimaLinha
        If Not IsEmpty(wsImp.Cells(r, 2).Value) Then
            If InStr(1, CStr(wsImp.Cells(r, 2).Value), "GERAIS DA PROPOSTA", vbTextCompare) > 0 Then
                linhaCondicoes = r
                Exit For
            End If
        End If
    Next r
    If linhaCondicoes = 0 Then Exit Sub

    ' Acha ultima linha com conteudo (fim da secao)
    linhaFimSecao = linhaCondicoes
    For r = ultimaLinha To linhaCondicoes Step -1
        If Not IsEmpty(wsImp.Cells(r, 1).Value) Or _
           Not IsEmpty(wsImp.Cells(r, 2).Value) Or _
           Not IsEmpty(wsImp.Cells(r, 3).Value) Then
            linhaFimSecao = r
            Exit For
        End If
    Next r

    ' Forca recomputacao de quebras automaticas
    Application.ScreenUpdating = False
    Set wnd = ActiveWindow
    originalView = wnd.View
    On Error Resume Next
    wnd.View = xlPageBreakPreview
    DoEvents
    On Error GoTo 0

    ' Verifica se ha quebra DENTRO da secao, ou quebra proxima ANTES
    breakNaSecao = False
    breakJustoAcima = False
    On Error Resume Next
    For Each hpb In wsImp.HPageBreaks
        If hpb.Location.Row > linhaCondicoes And hpb.Location.Row <= linhaFimSecao Then
            breakNaSecao = True
        End If
        ' Quebra "justo acima" = dentro das 2 linhas anteriores a CONDICOES
        If hpb.Location.Row >= linhaCondicoes - 1 And hpb.Location.Row <= linhaCondicoes + 1 Then
            breakJustoAcima = True
        End If
    Next hpb
    On Error GoTo 0

    ' Restaura view
    On Error Resume Next
    wnd.View = originalView
    On Error GoTo 0
    Application.ScreenUpdating = True

    ' Decisao:
    ' - Se quebra ja esta JUSTO ACIMA de CONDICOES -> nao faz nada (Excel ja
    '   paginou corretamente).
    ' - Senao, se houver quebra DENTRO da secao -> insere manual ANTES de
    '   CONDICOES para que a secao nao seja rachada.
    ' - Senao -> tudo fica na mesma pagina, nao faz nada.
    If Not breakJustoAcima And breakNaSecao Then
        On Error Resume Next
        wsImp.HPageBreaks.Add Before:=wsImp.Rows(linhaCondicoes)
        On Error GoTo 0
    End If
End Sub

VBA MACRO Módulo_Botão_Imprimir.bas 
in file: xl/vbaProject.bin - OLE stream: 'VBA/Módulo_Botão_Imprimir'

Sub ImprimirDuasAbasPDF()

    Dim abas() As String
    Dim caminho As Variant
    Dim numeroOrcamento As String
    Dim idx As Integer

    ' Captura número do orçamento na aba CAPA
    numeroOrcamento = Sheets("CAPA").Range("D5").Value

    ' Atualiza cabeçalho da aba Imp.Apres
    With Sheets("IMP_APRES").PageSetup
        .LeftHeader = ""
        .CenterHeader = ""
        .LeftHeader = Chr(10) & Chr(10) & "Orçamento nº: " & numeroOrcamento
    End With

    ' Atualiza cabeçalho da aba IMP_APRE_PRODU
    With Sheets("IMP_APRE_PRODU").PageSetup
        .LeftHeader = ""
        .CenterHeader = ""
        .LeftHeader = Chr(10) & Chr(10) & "Orçamento nº: " & numeroOrcamento
    End With

    ' Atualiza cabeçalho da aba Imp.Tabela
    With Sheets("Imp.Tabela").PageSetup
        .LeftHeader = ""
        .CenterHeader = ""
        .LeftHeader = Chr(10) & Chr(10) & "Orçamento nº: " & numeroOrcamento
    End With

    ' Inicializa o array com as abas fixas
ReDim abas(0 To 3)
abas(0) = "IMP_CAPA"
abas(1) = "IMP_APRES"
abas(2) = "IMP_APRE_PRODU"
abas(3) = "Imp.Tabela"
idx = 4


    ' Adiciona abas opcionais se necessário
    If chamaAbaSolar Then
        ReDim Preserve abas(0 To idx)
        abas(idx) = "SOLAR": idx = idx + 1
    End If

    If chamaAbaTrocador Then
        ReDim Preserve abas(0 To idx)
        abas(idx) = "TROCADOR": idx = idx + 1
    End If

    If chamaAbaEko Then
        ReDim Preserve abas(0 To idx)
        abas(idx) = "Trocador_Eko": idx = idx + 1
    End If

    ' Seleciona as abas
    ThisWorkbook.Sheets(abas).Select

    ' Salva PDF
    caminho = Application.GetSaveAsFilename( _
        InitialFileName:="Orçamento_" & numeroOrcamento & ".pdf", _
        FileFilter:="Arquivo PDF (*.pdf), *.pdf", _
        Title:="Salvar PDF do orçamento")

    If caminho <> False Then
        ActiveSheet.ExportAsFixedFormat _
            Type:=xlTypePDF, _
            Filename:=caminho, _
            Quality:=xlQualityStandard, _
            IncludeDocProperties:=True, _
            IgnorePrintAreas:=False, _
            OpenAfterPublish:=True

        ' Desagrupar as abas após salvar
        Sheets("IMP_CAPA").Select

    Else
        MsgBox "Operação cancelada.", vbInformation
    End If

End Sub

VBA MACRO Módulo_Aquecimento.bas 
in file: xl/vbaProject.bin - OLE stream: 'VBA/Módulo_Aquecimento'
- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
Sub CriarBlocoAquecimento()

    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim tblCadastro As ListObject
    Dim linha As ListRow, linha2 As ListRow
    Dim etapa As String
    Dim produto As String
    Dim qtde As Variant
    Dim textoProduto As String
    Dim blocoCriado As Boolean
    Dim primeiroProduto As String
    Dim segundoProduto As String
    Dim qtd1 As Variant, qtd2 As Variant
    Dim linkImagem As String
    Dim achouImagem As Boolean
    Dim linhaDestino As Long

    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")

    On Error Resume Next
    Set tbl = wsLinear.ListObjects("Tabela17")
    Set tblCadastro = wsDados.ListObjects("Cadastro")
    On Error GoTo 0

    If tbl Is Nothing Or tblCadastro Is Nothing Then
        MsgBox "Tabela17 ou Cadastro não encontrada.", vbExclamation
        Exit Sub
    End If

    textoProduto = ""
    primeiroProduto = ""
    segundoProduto = ""
    linhaDestino = proximaLinha
    blocoCriado = False

    chamaAbaTrocador = False
    chamaAbaSolar = False
    chamaAbaEko = False

    ' 1. Verificar produtos Ultra / Eko / Solar
    For Each linha In tbl.ListRows
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value

        If qtde > 0 And produto <> "" Then
            If InStr(1, LCase(produto), "tholz") > 0 Then chamaAbaTrocador = True
            If InStr(1, LCase(produto), "eko") > 0 Then chamaAbaEko = True
            If InStr(1, LCase(produto), "solar") > 0 Then chamaAbaSolar = True
        End If
    Next linha

    ' 2. Buscar dois produtos das etapas relevantes
    For Each linha In tbl.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value

        If (etapa = "aquecimento" Or etapa = "aquecimento solar") And qtde > 0 And produto <> "" Then
            If primeiroProduto = "" Then
                primeiroProduto = produto
                qtd1 = qtde
            ElseIf segundoProduto = "" And produto <> primeiroProduto Then
                segundoProduto = produto
                qtd2 = qtde
            End If
        End If
    Next linha

    ' Adicionar os dois produtos
    If primeiroProduto <> "" Then
        textoProduto = textoProduto & primeiroProduto & " – " & qtd1 & " un" & vbCrLf
        blocoCriado = True
    End If

    If segundoProduto <> "" Then
        textoProduto = textoProduto & segundoProduto & " – " & qtd2 & " un" & vbCrLf
        blocoCriado = True
    End If

    ' 3. Buscar bombas
    For Each linha In tbl.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value

        If (etapa = "bomba aquecimento" Or etapa = "bomba aquecimento solar") And qtde > 0 And produto <> "" Then
            textoProduto = textoProduto & produto & " – " & qtde & " un" & vbCrLf
            blocoCriado = True
        End If
    Next linha

    ' 4. Buscar imagem com prioridade: Aquecimento > Aquecimento solar
    achouImagem = False
    linkImagem = ""

    For Each linha In tbl.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value

        If etapa = "aquecimento" And qtde > 0 Then
            For Each linha2 In tblCadastro.ListRows
                If Trim(LCase(linha2.Range.Cells(1, tblCadastro.ListColumns("Descrição").Index).Value)) = Trim(LCase(produto)) Then
                    linkImagem = linha2.Range.Cells(1, tblCadastro.ListColumns("Link da Imagem").Index).Value
                    achouImagem = True
                    Exit For
                End If
            Next linha2
            If achouImagem Then Exit For
        End If
    Next linha

    If Not achouImagem Then
        For Each linha In tbl.ListRows
            etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
            produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
            qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value

            If etapa = "aquecimento solar" And qtde > 0 Then
                For Each linha2 In tblCadastro.ListRows
                    If Trim(LCase(linha2.Range.Cells(1, tblCadastro.ListColumns("Descrição").Index).Value)) = Trim(LCase(produto)) Then
                        linkImagem = linha2.Range.Cells(1, tblCadastro.ListColumns("Link da Imagem").Index).Value
                        achouImagem = True
                        Exit For
                    End If
                Next linha2
                If achouImagem Then Exit For
            End If
        Next linha
    End If

    ' 5. Escrever bloco na aba de impressão
    If blocoCriado Then
        ' Cabeçalho
        With wsImpressao.Cells(linhaDestino, 1)
            .Value = "AQUECIMENTO"
            .Font.Bold = True
            .Font.Size = 11
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With

        ' Texto principal
        With wsImpressao.Cells(linhaDestino, 3)
            Dim textoFinal As String
            textoFinal = vbCrLf & vbCrLf & "Conforme cálculos da próxima página"

            .Value = textoProduto & textoFinal
            .Font.Name = "Calibri"
            .Font.Size = 9
            .Font.Bold = False
            .WrapText = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop

            ' Aplicar formatação Rich Text no texto auxiliar
            Dim lenMain As Long: lenMain = Len(.Value) - Len(textoFinal)
            With .Characters(Start:=lenMain + 1, Length:=Len(textoFinal)).Font
                .Name = "Segoe UI"
                .Size = 7
                .Color = RGB(0, 112, 192)
                .Bold = False
                .Italic = False
            End With
        End With

        ' Imagem
        With wsImpressao.Cells(linhaDestino, 5)
            If linkImagem <> "" Then
                On Error Resume Next
                .FormulaLocal = "=IMAGEM(" & Chr(34) & linkImagem & Chr(34) & ")"
                If Err.Number <> 0 Then
                    .Value = "Imagem não encontrada"
                    Err.Clear
                End If
                On Error GoTo 0
            Else
                .Value = "Imagem não encontrada"
            End If
        End With

        ' Estética
        With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
            .Interior.Color = RGB(242, 242, 242)
            With .Borders(xlEdgeBottom)
                .LineStyle = xlContinuous
                .Color = RGB(180, 180, 180)
                .Weight = xlThin
            End With
        End With

        wsImpressao.Rows(linhaDestino).RowHeight = 80
        wsImpressao.Columns(10).ColumnWidth = 30
        wsImpressao.Columns(6).ColumnWidth = 20

        proximaLinha = linhaDestino + 1
    End If

End Sub

VBA MACRO Módulo_Revestimentos.bas 
in file: xl/vbaProject.bin - OLE stream: 'VBA/Módulo_Revestimentos'

Public Sub CriarBlocoRevestimento()

    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim linha As ListRow
    Dim etapa As String
    Dim descricao As String
    Dim quant As Variant
    Dim unidade As String
    Dim textoRevestimento As String, textoCantoneiras As String
    Dim quantRevestimento As Variant, quantCantoneiras As Variant
    Dim unidRevestimento As String, unidCantoneiras As String
    Dim textoFinal As String
    Dim enderecoImagem As String
    Dim linkImagem As String
    Dim blocoCriado As Boolean
    Dim linhaDestino As Long
    Dim linhaCadastro As Range
    Dim colDesc As Long, colLink As Long
    Dim produtoBusca As String

    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tbl = wsLinear.ListObjects("Tabela24")
    
    linhaDestino = proximaLinha
    blocoCriado = False

    ' Buscar dados na tabela da aba Linear
    For Each linha In tbl.ListRows
        etapa = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        descricao = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value))
        quant = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unidade = Trim(CStr(linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value))

        If Val(quant) > 0 And LCase(Left(descricao, 3)) <> "sem" Then
            If LCase(etapa) = "revestimentos" And descricao <> "" Then
                textoRevestimento = descricao
                quantRevestimento = quant
                unidRevestimento = unidade
            End If

            If LCase(etapa) = "cantoneiras" And descricao <> "" Then
                textoCantoneiras = descricao
                quantCantoneiras = quant
                unidCantoneiras = unidade
            End If
        End If
    Next linha

    ' Se tiver revestimento, montar bloco
    If textoRevestimento <> "" Then

        ' Cabeçalho coluna A
        With wsImpressao.Cells(linhaDestino, 1)
            .Value = "REVESTIMENTO"
            .Font.Bold = True
            .Font.Size = 11
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With

        ' === MONTAR TEXTO FINAL COLUNA C ===
        textoFinal = textoRevestimento & " – " & quantRevestimento & " " & unidRevestimento

        If textoCantoneiras <> "" Then
            textoFinal = textoFinal & vbNewLine & textoCantoneiras & " – " & quantCantoneiras & " " & unidCantoneiras
        End If

        ' === ESCREVER NA COLUNA C ===
        With wsImpressao.Cells(linhaDestino, 3)
            .Value = textoFinal
            .Font.Name = "Calibri"
            .Font.Size = 9
            .Font.Bold = False
            .WrapText = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With
     
        ' Procurar imagem do revestimento
        On Error Resume Next
        colDesc = wsDados.ListObjects("Cadastro").ListColumns("Descrição").Index
        colLink = wsDados.ListObjects("Cadastro").ListColumns("Link da Imagem").Index
        On Error GoTo 0

        linkImagem = ""
        If textoRevestimento <> "" Then
            For Each linhaCadastro In wsDados.ListObjects("Cadastro").DataBodyRange.Rows
                If Trim(UCase(linhaCadastro.Cells(1, colDesc).Value)) = Trim(UCase(textoRevestimento)) Then
                    linkImagem = linhaCadastro.Cells(1, colLink).Value
                    Exit For
                End If
            Next linhaCadastro
        End If

        ' === IMAGEM OU TEXTO NA COLUNA E ===
        With wsImpressao.Cells(linhaDestino, 5)
            If linkImagem <> "" Then
                .FormulaLocal = "=IMAGEM(" & Chr(34) & linkImagem & Chr(34) & ")"
            Else
                .Value = "Imagem não encontrada"
            End If
        End With

        ' Formatação visual
        With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
            .Interior.Color = RGB(242, 242, 242)
            With .Borders(xlEdgeBottom)
                .LineStyle = xlContinuous
                .Color = RGB(180, 180, 180)
                .Weight = xlThin
            End With
        End With

        wsImpressao.Rows(linhaDestino).RowHeight = 80
        wsImpressao.Columns(10).ColumnWidth = 30
        wsImpressao.Columns(6).ColumnWidth = 20

        proximaLinha = linhaDestino + 1
        blocoCriado = True
    End If

End Sub

VBA MACRO Módulo_Filtros.bas

Public Sub CriarBlocoConjuntoFiltrante()

    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim tblCadastro As ListObject
    Dim linha As ListRow
    Dim etapa As String
    Dim descricao As String
    Dim qtde As Variant
    Dim unid As String
    Dim eficiencia As Variant
    Dim linkImagem As String
    Dim linhaDestino As Long
    Dim cel As Range
    Dim textoProduto As String
    Dim blocoCriado As Boolean
    Dim colDesc As Long, colLink As Long

    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tbl = wsLinear.ListObjects("Tabela22")
    Set tblCadastro = wsDados.ListObjects("Cadastro")

    linhaDestino = proximaLinha
    blocoCriado = False

    ' Identificar colunas da tabela Cadastro
    On Error Resume Next
    colDesc = tblCadastro.ListColumns("Descrição").Index
    colLink = tblCadastro.ListColumns("Link da Imagem").Index
    On Error GoTo 0

    ' Percorre a Tabela22 para encontrar o conjunto filtrante
    For Each linha In tbl.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        descricao = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unid = Trim(linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value)

        On Error Resume Next
        eficiencia = linha.Range.Cells(1, tbl.ListColumns("Eficiência").Index).Value
        On Error GoTo 0

        If etapa = "conjunto filtrante" And qtde > 0 And descricao <> "" And LCase(descricao) <> "sem conjunto filtrante" Then

            ' === CABEÇALHO (COLUNA A) ===
            With wsImpressao.Cells(linhaDestino, 1)
                .Value = "CONJUNTO" & vbNewLine & "FILTRANTE"
                .Font.Bold = True
                .Font.Size = 11
                .HorizontalAlignment = xlLeft
                .VerticalAlignment = xlTop
            End With

            ' === TEXTO NA COLUNA C COM FORMATAÇÃO RICH TEXT ===
            textoProduto = descricao & vbNewLine & qtde & " " & unid

            With wsImpressao.Cells(linhaDestino, 3)
                .Value = textoProduto
                .Font.Name = "Calibri"
                .Font.Size = 9
                .Font.Bold = False
                .WrapText = True
                .HorizontalAlignment = xlLeft
                .VerticalAlignment = xlTop

                If Not IsEmpty(eficiencia) And eficiencia <> "" Then
                    Dim textoFinal As String
                    textoFinal = vbNewLine & vbNewLine & "Filtragem do volume total em " & eficiencia

                    .Value = .Value & textoFinal
                    Dim lenMain As Long: lenMain = Len(.Value) - Len(textoFinal)
                    With .Characters(Start:=lenMain + 1, Length:=Len(textoFinal)).Font
                        .Name = "Segoe UI"
                        .Size = 7
                        .Color = RGB(0, 112, 192) ' Azul claro
                        .Bold = False
                        .Italic = False
                    End With
                End If
            End With

            ' === BUSCAR IMAGEM NA TABELA CADASTRO ===
            linkImagem = ""
            For Each cel In tblCadastro.ListColumns(colDesc).DataBodyRange
                If Trim(UCase(cel.Value)) = Trim(UCase(descricao)) Then
                    linkImagem = cel.Offset(0, colLink - colDesc).Value
                    Exit For
                End If
            Next cel

            ' === ESCREVER IMAGEM NA COLUNA E ===
            With wsImpressao.Cells(linhaDestino, 5)
                If linkImagem <> "" Then
                    .FormulaLocal = "=IMAGEM(" & Chr(34) & linkImagem & Chr(34) & ")"
                Else
                    .Value = "Imagem não encontrada"
                End If
            End With

            ' === FORMATAÇÃO VISUAL ===
            With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
                .Interior.Color = RGB(242, 242, 242)
                With .Borders(xlEdgeBottom)
                    .LineStyle = xlContinuous
                    .Color = RGB(180, 180, 180)
                    .Weight = xlThin
                End With
            End With

            wsImpressao.Rows(linhaDestino).RowHeight = 80
            wsImpressao.Columns(10).ColumnWidth = 30
            wsImpressao.Columns(6).ColumnWidth = 20

            blocoCriado = True
            Exit For ' Apenas um conjunto filtrante por vez
        End If
    Next linha

    If blocoCriado Then
        proximaLinha = linhaDestino + 1
    End If

End Sub

Sub CriarBlocoModulosAutomacao()

    Dim wsImpressao As Worksheet

    Dim tbl26 As ListObject
    Dim tblCadastro As ListObject
    Dim linha As ListRow
    Dim etapa As String
    Dim produto As String
    Dim qtde As Variant
    Dim unid As Variant
    Dim textoProduto As String
    Dim descricaoContactor As String
    Dim totalContactor As Double
    Dim linkImagem As String
    Dim linhaDestino As Long
    Dim cel As Range
    Dim blocoCriado As Boolean
    Dim produtoParaImagem As String
    Dim descricaoSmart As String

    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tblCadastro = wsDados.ListObjects("Cadastro")
    Set tbl18 = wsLinear.ListObjects("Tabela18")
    Set tbl26 = wsLinear.ListObjects("Tabela26")

    linhaDestino = proximaLinha
    textoProduto = ""
    blocoCriado = False
    produtoParaImagem = ""
    descricaoSmart = ""

    ' --- Módulo da Tabela18 ---
    For Each linha In tbl18.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl18.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl18.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl18.ListColumns("Quant").Index).Value
        unid = linha.Range.Cells(1, tbl18.ListColumns("Unid").Index).Value

        If InStr(etapa, "módulo") > 0 And qtde > 0 And produto <> "" And Left(LCase(produto), 3) <> "sem" Then
            If InStr(LCase(produto), "smart") > 0 Then
                descricaoSmart = produto & " - " & qtde & " " & unid & Chr(10)
                produtoParaImagem = produto ' Prioriza imagem do Smart
                blocoCriado = True
            Else
                textoProduto = textoProduto & produto & " - " & qtde & " " & unid & Chr(10)
                If produtoParaImagem = "" Then produtoParaImagem = produto
                blocoCriado = True
            End If
        End If
    Next linha

    ' --- Módulos e Fontes da Tabela26 ---
    For Each linha In tbl26.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl26.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl26.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl26.ListColumns("Quant").Index).Value
        unid = linha.Range.Cells(1, tbl26.ListColumns("Unid").Index).Value

        If (etapa Like "*módulos de automação*" Or etapa Like "*módulo de automação solar*" Or etapa Like "*fonte*") _
            And qtde > 0 And produto <> "" And Left(LCase(produto), 3) <> "sem" Then
            If InStr(LCase(produto), "smart") > 0 Then
                descricaoSmart = produto & " - " & qtde & " " & unid & Chr(10)
                produtoParaImagem = produto ' Prioriza imagem do Smart
                blocoCriado = True
            Else
                textoProduto = textoProduto & produto & " - " & qtde & " " & unid & Chr(10)
                If produtoParaImagem = "" Then produtoParaImagem = produto
                blocoCriado = True
            End If
        End If
    Next linha

    ' --- Contactor (por último) ---
    totalContactor = 0
    descricaoContactor = ""
    For Each linha In tbl26.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl26.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl26.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl26.ListColumns("Quant").Index).Value

        If InStr(etapa, "contactor") > 0 And qtde > 0 And produto <> "" And Left(LCase(produto), 3) <> "sem" Then
            totalContactor = totalContactor + qtde
            descricaoContactor = produto
        End If
    Next linha

    ' --- Junta o Smart no topo se existir ---
    textoProduto = descricaoSmart & textoProduto

    If totalContactor > 0 And descricaoContactor <> "" Then
        textoProduto = textoProduto & descricaoContactor & " - " & totalContactor & " un" & Chr(10)
        blocoCriado = True
    End If

    If Not blocoCriado Then
        proximaLinha = linhaDestino
        Exit Sub
    End If

    ' --- Cabeçalho ---
    With wsImpressao.Cells(linhaDestino, 1)
        .Value = "MÓDULOS DE" & vbNewLine & "AUTOMAÇÃO"
        .Font.Bold = True
        .Font.Size = 11
        .HorizontalAlignment = xlLeft
        .VerticalAlignment = xlTop
    End With

    ' --- Descrição ---
    With wsImpressao.Cells(linhaDestino, 3)
        .Value = Trim(textoProduto)
        .Font.Name = "Calibri"
        .Font.Size = 9
        .Font.Bold = False
        .WrapText = True
        .HorizontalAlignment = xlLeft
        .VerticalAlignment = xlTop
    End With

    ' --- Imagem ---
    linkImagem = ""
    If produtoParaImagem <> "" Then
        For Each cel In tblCadastro.ListColumns("Descrição").DataBodyRange
            If Trim(LCase(cel.Value)) = Trim(LCase(produtoParaImagem)) Then
                linkImagem = cel.Offset(0, tblCadastro.ListColumns("Link da Imagem").Index - _
                    tblCadastro.ListColumns("Descrição").Index).Value
                Exit For
            End If
        Next cel
    End If

    If linkImagem <> "" Then
        On Error Resume Next
        wsImpressao.Cells(linhaDestino, 5).Formula = "=IMAGE(""" & linkImagem & """)"
        If Err.Number <> 0 Then
            wsImpressao.Cells(linhaDestino, 5).Value = "Imagem não encontrada"
            Err.Clear
        End If
        On Error GoTo 0
    Else
        wsImpressao.Cells(linhaDestino, 5).Value = "Imagem não encontrada"
    End If

    ' --- Estilo visual ---
    With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
        .Interior.Color = RGB(242, 242, 242)
        With .Borders(xlEdgeBottom)
            .LineStyle = xlContinuous
            .Color = RGB(180, 180, 180)
            .Weight = xlThin
        End With
    End With

    wsImpressao.Rows(linhaDestino).RowHeight = 80
    wsImpressao.Columns(10).ColumnWidth = 30
    wsImpressao.Columns(6).ColumnWidth = 20

    proximaLinha = linhaDestino + 1

End Sub

        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unid = linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value

            If LCase(Left(produto, 3)) <> "sem" Then

                If produtoImagem = "" Then produtoImagem = produto
                blocoCriado = True
                Exit For
            End If
        End If

            End If

        textoFinal = textoFinal & prod & " - " & dictDisjuntores(prod) & " un" & Chr(10)

                If produtoImagem = "" Then produtoImagem = produto
                blocoCriado = True
                Exit For
            End If
        End If
    Next linha

    ' Finaliza se encontrou produtos
    If blocoCriado Then
        Dim auxTexto As String
        auxTexto = "Conforme norma NBR 5410"
        textoFinal = Trim(textoFinal) & " " & auxTexto

Option Explicit

' ============================================================
'  CriarBlocoCasaDeMaquinas
'  - Monta o bloco "CASA DE MÁQUINAS" na aba IMP_APRE_PRODU
'  - Lista produtos de etapas de interesse (Tipo, Revestimento, Pintura, Ventilador, Tampa)
'  - Ignora produtos que começam com "Sem" ou "Serviço"
'  - Considera apenas itens com Quant > 0
'  - IMAGEM: prioridade Tipo (se existir e tiver link) -> fallback Tampa (se existir e tiver link)
'  - Mantém estética e comportamentos originais
' ============================================================
Sub CriarBlocoCasaDeMaquinas()

    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim tblCadastro As ListObject
    Dim linha As ListRow
    Dim etapa As String
    Dim produto As String
    Dim qtde As Variant
    Dim unid As String
    Dim linkImagem As String
    Dim linhaDestino As Long
    Dim textoProduto As String
    Dim termosBusca As Variant
    Dim imagemTipo As String            ' Mantido para compatibilidade com a sua lógica de inserção "=IMAGEM(" & imagemTipo & ")"
    Dim imagemDefinida As Boolean
    Dim cel As Range
    Dim i As Long

    ' --- Novas variáveis para PRIORIDADE da imagem ---
    Dim temTipo As Boolean, temTampa As Boolean
    Dim linkTipo As String, linkTampa As String
    Dim imagemSelecionada As String

    ' Referências
    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tbl = wsLinear.ListObjects("Tabela19")
    Set tblCadastro = wsDados.ListObjects("Cadastro")

    ' Controle de destino e inicializações
    linhaDestino = proximaLinha
    textoProduto = ""
    termosBusca = Array("Tipo", "Revestimento", "Pintura", "Ventilador", "Tampa")
    imagemDefinida = False
    temTipo = False
    temTampa = False
    linkTipo = ""
    linkTampa = ""
    imagemTipo = ""                     ' será preenchida ao final respeitando a prioridade

    ' ============================
    ' Coleta dados na Tabela19
    ' ============================
    For Each linha In tbl.ListRows

        ' Leitura das colunas de interesse
        etapa = Trim(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value)
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unid = Trim(linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value)

        ' Só processa se a etapa não for vazia
        If etapa <> "" Then

            ' Ignora "Sem..." e "Serviço..."
            If Left$(LCase$(produto), 3) = "sem" Or Left$(LCase$(produto), 7) = "serviço" Then GoTo ProximoItem

            ' Só considera para listagem e captura de imagem quando qtde > 0
            If qtde > 0 Then

                ' Lista apenas quando a etapa está entre as de interesse (termosBusca)
                For i = LBound(termosBusca) To UBound(termosBusca)
                    If InStr(1, etapa, termosBusca(i), vbTextCompare) > 0 Then

                        ' Monta a descrição consolidada dos produtos
                        textoProduto = textoProduto & produto & " - " & qtde & " " & unid & vbCrLf

                        ' -------------------------------------------------------
                        ' Candidatos para IMAGEM:
                        ' 1) TIPO: prioridade máxima (captura primeiro link válido)
                        ' 2) TAMPA: fallback (caso Tipo não exista ou não tenha link)
                        ' -------------------------------------------------------

                        ' Se a etapa é "Tipo" e ainda não temos um link de Tipo, tenta localizar o link na Cadastro
                        If Not temTipo And etapa Like "*Tipo*" Then
                            linkImagem = BuscarLinkImagem(produto, tblCadastro)
                            If Len(Trim$(linkImagem)) > 0 Then
                                temTipo = True
                                linkTipo = linkImagem
                            End If
                        End If

                        ' Se a etapa é "Tampa" e ainda não temos um link de Tampa, tenta localizar o link na Cadastro
                        If Not temTampa And etapa Like "*Tampa*" Then
                            linkImagem = BuscarLinkImagem(produto, tblCadastro)
                            If Len(Trim$(linkImagem)) > 0 Then
                                temTampa = True
                                linkTampa = linkImagem
                            End If
                        End If

                        Exit For ' já tratou essa linha por fazer parte das etapas de interesse
                    End If
                Next i

            End If
        End If

ProximoItem:
    Next linha

    ' =========================================
    ' Seleção final da imagem respeitando a prioridade
    ' Tipo (se encontrado) -> senão Tampa (se encontrado)
    ' =========================================
    If Not imagemDefinida Then
        If temTipo And Len(Trim$(linkTipo)) > 0 Then
            imagemSelecionada = linkTipo
            imagemDefinida = True
        ElseIf temTampa And Len(Trim$(linkTampa)) > 0 Then
            imagemSelecionada = linkTampa
            imagemDefinida = True
        End If

        ' Mantém o mesmo formato string entre aspas para a função IMAGEM()
        If imagemDefinida Then
            imagemTipo = Chr$(34) & imagemSelecionada & Chr$(34)
        End If
    End If

    ' =========================================
    ' Escrita no relatório (IMP_APRE_PRODU)
    ' Mantém a estética e comportamentos originais
    ' =========================================
    If textoProduto <> "" Then

        ' Cabeçalho (coluna A)
        With wsImpressao.Cells(linhaDestino, 1)
            .Value = "CASA DE MÁQUINAS"
            .Font.Bold = True
            .Font.Size = 11
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With

        ' Descrição consolidada (coluna C)
        With wsImpressao.Cells(linhaDestino, 3)
            .Value = Trim$(textoProduto)
            .Font.Name = "Calibri"
            .Font.Size = 9
            .Font.Bold = False
            .WrapText = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
        End With

        ' Imagem (coluna E), usando a string imagemTipo já formatada com aspas
        On Error Resume Next
        If imagemTipo <> "" Then
            wsImpressao.Cells(linhaDestino, 5).FormulaLocal = "=IMAGEM(" & imagemTipo & ")"
        Else
            wsImpressao.Cells(linhaDestino, 5).Value = "Imagem não encontrada"
        End If
        On Error GoTo 0

        ' Estética da linha
        With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
            .Interior.Color = RGB(242, 242, 242)
            With .Borders(xlEdgeBottom)
                .LineStyle = xlContinuous
                .Color = RGB(180, 180, 180)
                .Weight = xlThin
            End With
        End With

        ' Altura da linha e larguras de colunas (mantidas)
        wsImpressao.Rows(linhaDestino).RowHeight = 80
        wsImpressao.Columns(10).ColumnWidth = 40
        wsImpressao.Columns(6).ColumnWidth = 30

        ' Avança a próxima linha do relatório
        proximaLinha = linhaDestino + 1

    Else
        ' Sem itens listados: não avança a linha
        proximaLinha = linhaDestino
    End If

End Sub


' ============================================================
'  BuscarLinkImagem
'  - Procura na tabela "Cadastro" (aba Dados) o link na coluna
'    "Link da Imagem" correspondente à "Descrição" informada.
'  - Retorna "" se não encontrar.
' ============================================================
Private Function BuscarLinkImagem(ByVal descricao As String, ByRef tblCadastro As ListObject) As String
    Dim cel As Range
    Dim idxDesc As Long, idxLink As Long

    idxDesc = tblCadastro.ListColumns("Descrição").Index
    idxLink = tblCadastro.ListColumns("Link da Imagem").Index

    For Each cel In tblCadastro.ListColumns("Descrição").DataBodyRange
        If Trim$(LCase$(cel.Value)) = Trim$(LCase$(descricao)) Then
            BuscarLinkImagem = cel.Offset(0, idxLink - idxDesc).Value
            Exit Function
        End If
    Next cel

    ' Se não encontrou, retorna vazio
    BuscarLinkImagem = ""
End Function

Sub CriarBlocoRalosNormatizados()

    ' === Declarações ===
    Dim wsLinear As Worksheet
    Dim wsImpressao As Worksheet
    Dim wsDados As Worksheet
    Dim tbl As ListObject
    Dim tblCadastro As ListObject
    Dim linha As ListRow
    Dim etapa As String, produto As String, qtde As Variant, unid As String
    Dim linkImagem As String, enderecoImagem As String
    Dim linhaDestino As Long, textoDescricao As String
    Dim cel As Range, blocoCriado As Boolean, primeiraImagem As Boolean

    ' === Definições de objetos ===
    Set wsLinear = ThisWorkbook.Sheets("Linear")
    Set wsImpressao = ThisWorkbook.Sheets("IMP_APRE_PRODU")
    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tbl = wsLinear.ListObjects("Tabela6")
    Set tblCadastro = wsDados.ListObjects("Cadastro")

    linhaDestino = proximaLinha
    textoDescricao = ""
    enderecoImagem = ""
    primeiraImagem = True
    blocoCriado = False

    ' === Percorre a Tabela6 buscando etapas: ralos de fundo, ralos de parede ou skimmer ===
    For Each linha In tbl.ListRows
        etapa = Trim(LCase(linha.Range.Cells(1, tbl.ListColumns("Etapa").Index).Value))
        produto = Trim(linha.Range.Cells(1, tbl.ListColumns("Descrição").Index).Value)
        qtde = linha.Range.Cells(1, tbl.ListColumns("Quant").Index).Value
        unid = linha.Range.Cells(1, tbl.ListColumns("Unid").Index).Value

        If etapa Like "*ralos de fundo*" Or etapa Like "*ralos de parede*" Or etapa Like "*skimmer*" Then
            If qtde > 0 And produto <> "" And LCase(produto) <> "sem" Then

                ' === Adiciona à descrição ===
                textoDescricao = textoDescricao & produto & Chr(10) & qtde & " " & unid & Chr(10)

                ' === Pega imagem do primeiro produto válido ===
                If primeiraImagem Then
                    For Each cel In tblCadastro.ListColumns("Descrição").DataBodyRange
                        If Trim(LCase(cel.Value)) = Trim(LCase(produto)) Then
                            linkImagem = cel.Offset(0, tblCadastro.ListColumns("Link da Imagem").Index - _
                                tblCadastro.ListColumns("Descrição").Index).Value
                            If linkImagem <> "" Then
                                enderecoImagem = Chr(34) & linkImagem & Chr(34)
                                primeiraImagem = False
                                Exit For
                            End If
                        End If
                    Next cel
                End If

                blocoCriado = True
            End If
        End If
    Next linha

    ' === Se encontrou produtos válidos, monta o bloco ===
    If blocoCriado Then

        ' === Cabeçalho na coluna A ===
        With wsImpressao.Cells(linhaDestino, 1)
            .Value = "RALOS" & Chr(10) & "NORMATIZADOS"
            .Font.Name = "Calibri"
            .Font.Size = 11
            .Font.Bold = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop
            .WrapText = True
        End With

        ' === Texto com espaçamento e texto auxiliar ===
        textoDescricao = Trim(textoDescricao) & vbNewLine & "Conforme norma NBR 10339"

        ' === Coluna C com descrição formatada ===
        With wsImpressao.Cells(linhaDestino, 3)
            .Value = textoDescricao
            .Font.Name = "Calibri"
            .Font.Size = 9
            .Font.Bold = False
            .WrapText = True
            .HorizontalAlignment = xlLeft
            .VerticalAlignment = xlTop

            ' Formatar o texto auxiliar
            Dim textoAux As String
            textoAux = "Conforme norma NBR 10339"
            Dim posInicio As Long
            posInicio = Len(.Value) - Len(textoAux) + 1

            With .Characters(Start:=posInicio, Length:=Len(textoAux)).Font

                .Size = 7
                .Color = RGB(0, 112, 192) ' Azul claro
                .Italic = False
                .Bold = False
            End With
        End With

        ' === Imagem ou aviso ===
        If enderecoImagem <> "" Then
            wsImpressao.Cells(linhaDestino, 5).FormulaLocal = "=IMAGEM(" & enderecoImagem & ")"
        Else
            wsImpressao.Cells(linhaDestino, 5).Value = "Imagem não encontrada"

        With wsImpressao.Range(wsImpressao.Cells(linhaDestino, 1), wsImpressao.Cells(linhaDestino, 5))
            .Interior.Color = RGB(242, 242, 242)

                .LineStyle = xlContinuous
                .Color = RGB(180, 180, 180)
                .Weight = xlThin
            End With
        End With

        wsImpressao.Rows(linhaDestino).RowHeight = 80
        wsImpressao.Columns(10).ColumnWidth = 30
        wsImpressao.Columns(6).ColumnWidth = 20

        ' === Próxima linha ===
        proximaLinha = linhaDestino + 1
    Else
        proximaLinha = linhaDestino
    End If

Private Function SanitizarNomeArquivo(link As String) As String
    On Error GoTo TratarErro
    Dim nome As String

    nome = Replace(link, "https://", "")
    nome = Replace(nome, "http://", "")
    nome = Replace(nome, "/", "_")
    nome = Replace(nome, "\", "_")
    nome = Replace(nome, ":", "_")
    nome = Replace(nome, "*", "_")
    nome = Replace(nome, "?", "_")
    nome = Replace(nome, """", "_")
    nome = Replace(nome, "<", "_")
    nome = Replace(nome, ">", "_")
    nome = Replace(nome, "|", "_")

    If Trim(nome) = "" Then nome = "imagem_padrao.jpg"
    SanitizarNomeArquivo = nome
    Exit Function

TratarErro:
    SanitizarNomeArquivo = "imagem_padrao.jpg"
End Function
Function ExtrairNomeDoArquivo(link As String) As String
    Dim partes() As String
    partes = Split(link, "/")
    ExtrairNomeDoArquivo = partes(UBound(partes)) ' Última parte do link
End Function

Private Sub lstResultados_Click()
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim linha As ListRow
    Dim codigo As String
    Dim valorImagem As String
    Dim nomeArquivo As String
    Dim caminhoLocal As String
    Dim pastaBase As String

    Me.btnEditar.Enabled = True

    codigo = Trim(Me.lstResultados.List(Me.lstResultados.ListIndex, 0))
    Set ws = ThisWorkbook.Sheets("Dados")
    Set tbl = ws.ListObjects("Cadastro")

    ' Busca link da imagem pelo código
    For Each linha In tbl.ListRows
        If Trim(linha.Range(1, tbl.ListColumns("Cod").Index).Value) = codigo Then
            valorImagem = Trim(linha.Range(1, tbl.ListColumns("Link da Imagem").Index).Value)
            Exit For
        End If
    Next linha

    Debug.Print "Link encontrado: " & valorImagem

    ' Se houver imagem válida
    If valorImagem <> "" And LCase(valorImagem) <> "sem imagem" Then
        nomeArquivo = ExtrairNomeDoArquivo(valorImagem)
        pastaBase = ThisWorkbook.Path

        ' Verifica se a planilha está salva localmente
        If Len(pastaBase) = 0 Or InStr(pastaBase, ":") = 0 Then

            MsgBox "Caminho inválido. O arquivo não está disponível localmente: " & caminhoLocal, vbExclamation
            Me.imgPesquisa.Picture = Nothing
            Exit Sub
        End If

        If Dir(caminhoLocal) <> "" Then
            Debug.Print "Imagem encontrada localmente"
            On Error Resume Next
            Me.imgPesquisa.Picture = LoadPicture(caminhoLocal)
            If Err.Number <> 0 Then
                Debug.Print "Erro ao carregar imagem local: " & Err.Description
                Me.imgPesquisa.Picture = Nothing
            End If
            On Error GoTo 0
        Else
        
            If BaixarImagem(valorImagem, caminhoLocal) Then
                Debug.Print "Imagem baixada com sucesso"
                On Error Resume Next
                Me.imgPesquisa.Picture = LoadPicture(caminhoLocal)

                    Debug.Print "Erro ao carregar imagem após download: " & Err.Description
                    Me.imgPesquisa.Picture = Nothing
                End If
                On Error GoTo 0
            Else
                Debug.Print "Erro ao baixar imagem - modo offline"
                Me.imgPesquisa.Picture = Nothing
            End If
        End If
    Else
        Debug.Print "Link da imagem vazio ou inválido"
        Me.imgPesquisa.Picture = Nothing
    End If

    If tbl Is Nothing Then Exit Sub

    For Each cel In tbl.ListColumns(nomeColuna).DataBodyRange

    cb.Clear
    For Each valorOriginal In dict.Items
        cb.AddItem valorOriginal
    Next valorOriginal

Private Sub CarregarImagem(link As String, imgCtrl As Image)
    Dim caminhoLocal As String
    Dim nomeArquivo As String

    On Error GoTo TratarErro

    If Trim(link) = "" Then
        Set imgCtrl.Picture = Nothing
        imgCtrl.ControlTipText = "Sem imagem"
        Exit Sub
    End If

    nomeArquivo = SanitizarNomeArquivo(link)
    caminhoLocal = ThisWorkbook.Path & "\bitmap\" & nomeArquivo

    ' Verifica se a imagem já existe localmente
    If Dir(caminhoLocal) <> "" Then
        imgCtrl.Picture = LoadPicture(caminhoLocal)
        imgCtrl.ControlTipText = "Imagem local carregada"
        Exit Sub
    End If

    ' Tenta baixar a imagem se estiver online
    If TestarConexaoInternet() Then
        If BaixarImagem(link, caminhoLocal) Then
            imgCtrl.Picture = LoadPicture(caminhoLocal)
            imgCtrl.ControlTipText = "Imagem baixada"
        Else
            Set imgCtrl.Picture = Nothing
            imgCtrl.ControlTipText = "Erro ao baixar imagem"
        End If
    Else
        Set imgCtrl.Picture = Nothing
        imgCtrl.ControlTipText = "Offline - imagem não disponível"
    End If
    Exit Sub

TratarErro:
    Set imgCtrl.Picture = Nothing
    imgCtrl.ControlTipText = "Erro ao carregar imagem"
End Sub

'-----------------------------------------------------------
' Baixa a imagem online para o cache local
'-----------------------------------------------------------
Private Function BaixarImagem(link As String, destinoLocal As String) As Boolean
    On Error GoTo erro
    Dim http As Object
    Dim ado As Object
    Dim arr() As Byte

    Set http = CreateObject("MSXML2.XMLHTTP")
    http.Open "GET", link, False
    http.Send

    If http.Status = 200 Then
        arr = http.responseBody
        Set ado = CreateObject("ADODB.Stream")
        ado.Type = 1
        ado.Open
        ado.Write arr
        ado.SaveToFile destinoLocal, 2
        ado.Close
        BaixarImagem = True
    Else
        BaixarImagem = False
    End If
    Exit Function

erro:
    BaixarImagem = False
End Function

Private Sub btnPesquisar_Click()
    Dim ws As Worksheet
    Dim tbl As ListObject
    Dim i As Long
    Dim match As Boolean
    Dim critDescricao As String, critCodigo As String
    Dim critMarca As String, critTipo As String
    Dim critCategoria As String, critGrupo As String

    Me.lstResultados.Clear

    critDescricao = LCase(Trim(Me.txtPesquisaDescricao.Value))
    critCodigo = LCase(Trim(Me.txtPesquisaCodigo.Value))
    critMarca = LCase(Trim(Me.cboPesquisaMarca.Value))
    critTipo = LCase(Trim(Me.cboPesquisaTipo.Value))
    critCategoria = LCase(Trim(Me.cboPesquisaCategoria.Value))
    critGrupo = LCase(Trim(Me.cboPesquisaGrupo.Value))

    Set ws = ThisWorkbook.Sheets("Dados")
    Set tbl = ws.ListObjects("Cadastro")

    For i = 1 To tbl.ListRows.Count
        match = True

        If critDescricao <> "" Then
            If InStr(1, LCase(tbl.DataBodyRange(i, tbl.ListColumns("Descrição").Index).Value), critDescricao) = 0 Then
                match = False
            End If
        End If
        If critCodigo <> "" Then
            If InStr(1, LCase(tbl.DataBodyRange(i, tbl.ListColumns("Cod").Index).Value), critCodigo) = 0 Then
                match = False
            End If
        End If
        If critMarca <> "" Then
            If LCase(tbl.DataBodyRange(i, tbl.ListColumns("Marca").Index).Value) <> critMarca Then
                match = False
            End If
        End If
        If critTipo <> "" Then
            If LCase(tbl.DataBodyRange(i, tbl.ListColumns("Tipo").Index).Value) <> critTipo Then
                match = False
            End If
        End If
        If critCategoria <> "" Then
            If LCase(tbl.DataBodyRange(i, tbl.ListColumns("Categoria").Index).Value) <> critCategoria Then
                match = False
            End If
        End If
        If critGrupo <> "" Then
            If LCase(tbl.DataBodyRange(i, tbl.ListColumns("Grupo").Index).Value) <> critGrupo Then
                match = False
            End If
        End If

        If match Then
            Me.lstResultados.AddItem
            Me.lstResultados.List(Me.lstResultados.ListCount - 1, 0) = tbl.DataBodyRange(i, tbl.ListColumns("Cod").Index).Value
            Me.lstResultados.List(Me.lstResultados.ListCount - 1, 1) = tbl.DataBodyRange(i, tbl.ListColumns("Descrição").Index).Value
            Me.lstResultados.List(Me.lstResultados.ListCount - 1, 2) = tbl.DataBodyRange(i, tbl.ListColumns("Unid").Index).Value
            Me.lstResultados.List(Me.lstResultados.ListCount - 1, 3) = tbl.DataBodyRange(i, tbl.ListColumns("Marca").Index).Value
            Me.lstResultados.List(Me.lstResultados.ListCount - 1, 4) = "R$ " & Format(tbl.DataBodyRange(i, tbl.ListColumns("R$ Custo").Index).Value, "#,##0.00")
        End If
    Next i

    If Me.lstResultados.ListCount = 0 Then
        MsgBox "Nenhum registro encontrado.", vbInformation
    End If
End Sub

Private Sub btnImport_Click()
    Dim wbImport As Workbook
    Dim wsImport As Worksheet
    Dim tbl As ListObject
    Dim filePath As String
    Dim lastRowImport As Long
    Dim i As Long
    Dim cod As String, descricao As String, marca As String, unidade As String
    Dim custo As Double
    Dim linhaExistente As ListRow
    Dim wsDados As Worksheet
    Dim linha As ListRow
    Dim senha As String

    ' Caminho do arquivo
    filePath = Application.GetOpenFilename("Arquivos Excel (*.xls; *.xlsx), *.xls; *.xlsx", , "Selecione a planilha do Sankhya")
    If filePath = "Falso" Then Exit Sub

    Application.ScreenUpdating = False
    Application.EnableEvents = False

    ' Define aba de destino
    Set wsDados = ThisWorkbook.Sheets("Dados")

    ' Verifica se aba está protegida
    If wsDados.ProtectContents = True Then
        senha = InputBox("A aba 'Dados' está protegida. Por favor, insira a senha para continuar:", "Senha da aba Dados")
        If senha = "" Then
            MsgBox "Importação cancelada. Senha não informada.", vbExclamation
            GoTo Finalizar
        End If
        On Error Resume Next
        wsDados.Unprotect Password:=senha
        If wsDados.ProtectContents = True Then
            MsgBox "Senha incorreta. A importação foi cancelada.", vbCritical
            GoTo Finalizar
        End If
        On Error GoTo 0
    End If

    ' Abre o arquivo de importação
    Set wbImport = Workbooks.Open(filePath)
    Set wsImport = wbImport.Sheets(1) ' Primeira aba
    lastRowImport = wsImport.Cells(wsImport.Rows.Count, "A").End(xlUp).Row

    ' Define a tabela
    Set tbl = wsDados.ListObjects("Cadastro") ' Nome exato da TABELA

    ' Percorre os dados a partir da linha 4
    For i = 4 To lastRowImport
        cod = Trim(wsImport.Cells(i, "A").Value)
        descricao = Trim(wsImport.Cells(i, "B").Value)
        marca = Trim(wsImport.Cells(i, "C").Value)
        unidade = Trim(wsImport.Cells(i, "D").Value)
        custo = Val(wsImport.Cells(i, "I").Value)

        If cod <> "" Then
            ' Procura o código na tabela
            Set linhaExistente = Nothing
            For Each linha In tbl.ListRows
                If linha.Range(1, tbl.ListColumns("Cod").Index).Value = cod Then
                    Set linhaExistente = linha
                    Exit For
                End If
            Next linha

            If Not linhaExistente Is Nothing Then
                ' Atualiza dados existentes
                With linhaExistente
                    .Range(1, tbl.ListColumns().Index).Value = descricao
                    .Range(1, tbl.ListColumns("Marca").Index).Value = marca
                    .Range(1, tbl.ListColumns("Unid").Index).Value = unidade
                    .Range(1, tbl.ListColumns("R$ Custo").Index).Value = custo
                End With
            Else
                ' Adiciona nova linha na tabela
                Set linha = tbl.ListRows.Add
                With linha
                    .Range(1, tbl.ListColumns("Cod").Index).Value = cod
                    .Range(1, tbl.ListColumns().Index).Value = descricao
                    .Range(1, tbl.ListColumns("Marca").Index).Value = marca
                    .Range(1, tbl.ListColumns("Unid").Index).Value = unidade
                    .Range(1, tbl.ListColumns("R$ Custo").Index).Value = custo
                End With
            End If
        End If
    Next i

    wbImport.Close SaveChanges:=False
    MsgBox "Importação concluída com sucesso!", vbInformation

Finalizar:
    Application.ScreenUpdating = True
    Application.EnableEvents = True
End Sub

Function CapitalizarCadaPalavra(ByVal texto As String) As String
    Dim partes() As String
    Dim i As Long
    texto = LCase(Trim(texto))
    
    If texto = "" Then
        CapitalizarCadaPalavra = ""
        Exit Function
    End If

    partes = Split(texto, " ")
    For i = 0 To UBound(partes)
        If Len(partes(i)) > 0 Then
            partes(i) = UCase(Left(partes(i), 1)) & Mid(partes(i), 2)
        End If
    Next i

    CapitalizarCadaPalavra = Join(partes, " ")
End Function

Public Function TestarConexaoInternet() As Boolean
    On Error GoTo Falha
    Dim http As Object
    Set http = CreateObject("MSXML2.XMLHTTP")
    http.Open "HEAD", "https://www.google.com", False
    http.Send
    TestarConexaoInternet = (http.Status = 200)
    Exit Function

Falha:
    TestarConexaoInternet = False
End Function

Sub AtualizarFiltroAutomatico()

    Dim wsCapa As Worksheet, wsLinear As Worksheet, wsDados As Worksheet, wsAux As Worksheet
    Dim tblLinear As ListObject, tblCadastro As ListObject
    Dim i As Long
    Dim volume As Double, vazaoIdeal As Double, vazao As Double
    Dim menorVazao As Double: menorVazao = 9999
    Dim melhorFiltro As String
    Dim filtroValido As Boolean
    Dim listaAux As Range, cel As Range

    Set wsCapa = Sheets("CAPA")
    Set wsLinear = Sheets("Linear")
    Set wsDados = Sheets("Dados")
    Set wsAux = Sheets("Auxiliar")
    Set tblLinear = wsLinear.ListObjects("Tabela22")
    Set tblCadastro = wsDados.ListObjects("Cadastro")

    ' Volume da piscina
    volume = wsCapa.Range("E16").Value
    If volume <= 0 Then Exit Sub

    vazaoIdeal = volume / 3.7

    ' Procurar o melhor filtro com base em Grupo = Filtros, Categoria = Piscinas, Ativo = Sim
    For i = 1 To tblCadastro.ListRows.Count
        If LCase(Trim(tblCadastro.DataBodyRange(i, tblCadastro.ListColumns("Grupo").Index).Value)) = "filtros" _
        And LCase(Trim(tblCadastro.DataBodyRange(i, tblCadastro.ListColumns("Categoria").Index).Value)) = "piscinas" _
        And LCase(Trim(tblCadastro.DataBodyRange(i, tblCadastro.ListColumns("Ativo").Index).Value)) = "sim" Then

            vazao = Val(Replace(tblCadastro.DataBodyRange(i, tblCadastro.ListColumns("Vazão m³/h").Index).Text, ",", "."))

            If vazao >= vazaoIdeal And vazao < menorVazao Then
                menorVazao = vazao
                melhorFiltro = tblCadastro.DataBodyRange(i, tblCadastro.ListColumns("Descrição").Index).Value
            End If
        End If
    Next i

    If melhorFiltro = "" Then
        MsgBox "? Nenhum filtro encontrado com vazão = " & Round(vazaoIdeal, 2) & " m³/h", vbExclamation
        Exit Sub
    End If

    ' Verificar se está na lista de validação
    Set listaAux = wsAux.Range("P2:P100")
    filtroValido = False
    For Each cel In listaAux
        If Trim(cel.Value) = Trim(melhorFiltro) Then
            filtroValido = True
            Exit For
        End If
    Next cel

    If Not filtroValido Then
        MsgBox "? O filtro '" & melhorFiltro & "' não está na lista de validação da aba Auxiliar!", vbExclamation
        Exit Sub
    End If

    ' Aplicar na Tabela22 onde Etapa = Conjunto filtrante
    For i = 1 To tblLinear.ListRows.Count
        If StrComp(Trim(tblLinear.ListRows(i).Range(1, 2).Value), "Conjunto filtrante", vbTextCompare) = 0 Then
            Application.EnableEvents = False
            With tblLinear.ListRows(i).Range(1, 3)
                .ClearContents
                .Value = melhorFiltro
            End With
            Application.EnableEvents = True
            
            Exit Sub
        End If
    Next i

    MsgBox "? Etapa 'Conjunto filtrante' não encontrada na Tabela22.", vbExclamation

End Sub
Public Sub AtualizarTuboDoFiltro()

    Dim wsLinear As Worksheet
    Dim wsDados As Worksheet
    Dim tblLinear As ListObject
    Dim tblCadastro As ListObject
    Dim nomeFiltro As String
    Dim diametroTubo As Variant
    Dim descricaoTubo As String
    Dim i As Long

    Set wsDados = ThisWorkbook.Sheets("Dados")
    Set tblLinear = wsLinear.ListObjects("Tabela22")
    Set tblCadastro = wsDados.ListObjects("Cadastro")

    ' 1. Encontrar o nome do filtro na Tabela22
    For i = 1 To tblLinear.ListRows.Count
        If StrComp(Trim(tblLinear.ListRows(i).Range(1, 2).Value), "Conjunto filtrante", vbTextCompare) = 0 Then
            nomeFiltro = Trim(tblLinear.ListRows(i).Range(1, 3).Value)
            Exit For
        End If
    Next i

    If nomeFiltro = "" Then Exit Sub

    ' 2. Buscar o filtro na Tabela Cadastro e pegar o tubo
    For i = 1 To tblCadastro.ListRows.Count

    descricaoTubo = "Tubos conexões e registros p/ sist. filtragem " & diametroTubo & "mm"

            Application.EnableEvents = True

VBA MACRO MóduloParcelas.bas 
in file: xl/vbaProject.bin - OLE stream: 'VBA/MóduloParcelas'
- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - 
Option Explicit

Sub GerarTabelaParcelas()

    ' ============================================================
    ' GERAR TABELA DE PARCELAS (TabelaParcelas) NA ABA "Linear"
    ' ============================================================
    ' Lê na Tabela86 (aba Linear):
    ' - Forma de pagamento
    ' - Previsão de início / término
    ' - Desconto para pagamento antecipado
    '
    ' Lê na tabela "Parcelas" (aba CALCULO_PISCINA):
    ' - Entrada em %
    ' - Qtd Parcelas
    ' - Prazo 1ª parcela dias
    ' - Intervalo dias
    ' - Tipo
    ' - Imposto %  (NOVA COLUNA)
    '
    ' Tabela70 (aba Linear) NÃO tem cabeçalho e tem 2 colunas:
    ' Coluna 1: rótulo (SUBTOTAL:, IMPOSTOS:, DESCONTO %:, TOTAL GERAL:)
    ' Coluna 2: valor (pode ser fórmula)
    '
    ' NOVA LÓGICA DO IMPOSTO:
    ' - O imposto da Tabela70 é regravado como fórmula:
    '   IMPOSTOS = SUBTOTAL * (Imposto% / 100)
    ' - Para formas sem imposto: Imposto% = 0  -> fórmula resulta 0
    ' - Assim o TOTAL GERAL passa a recalcular corretamente pela própria planilha.
    '
    ' AJUSTE DE CENTAVOS:
    ' - Corrige a última parcela para garantir soma exata = TOTAL GERAL
    ' ============================================================

    Dim ws As Worksheet, wsParametros As Worksheet
    Dim tbl As ListObject, tbl86 As ListObject, tbl70 As ListObject
    Dim tblParcelas As ListObject
    Dim rngDestino As Range

    Dim nomeForma As String
    Dim linhaForma As Long
    Dim i As Long

    Dim entradaPct As Double, valorTotal As Double
    Dim valorEntrada As Double, valorParcela As Double
    Dim numParcelas As Long, prazo As Long
    Dim intervalo As Variant

    Dim dataParcela As Date, dataInicio As Date, dataFinal As Date
    Dim tipoForma As String
    Dim dataBase As Date
    Dim linhaBase As Long
    Dim celula As Range
    Dim linhaTotal As ListRow
    Dim nomeLinha As String

    Dim descontoValor As Double
    Dim celulaDesconto As Range

    ' Referências dentro da Tabela70
    Dim celulaSubtotal As Range
    Dim celulaImpostos As Range
    Dim celulaTotalGeral As Range

    ' Imposto % vindo da tabela Parcelas
    Dim impostoPct As Double

    ' Ajuste de soma das parcelas
    Dim somaParcelas As Double
    Dim diferenca As Double

    ' Objeto da tabela criada
    Dim tblCriada As ListObject

    ' ============================================================
    ' DEFINIÇÃO DAS PLANILHAS / TABELAS
    ' ============================================================
    Set ws = ThisWorkbook.Sheets("Linear")
    Set wsParametros = ThisWorkbook.Sheets("CALCULO_PISCINA")

    Set tbl86 = ws.ListObjects("Tabela86")
    Set tbl70 = ws.ListObjects("Tabela70")
    Set tblParcelas = wsParametros.ListObjects("Parcelas")

    ' ============================================================
    ' 1) LER DADOS NA TABELA86
    ' ============================================================
    nomeForma = vbNullString
    descontoValor = 0

    For Each celula In tbl86.ListColumns(1).DataBodyRange

        Select Case Trim(celula.Value)

            Case "Forma de pagamento:"
                nomeForma = Trim(celula.Offset(0, 1).Value)

            Case "Previsão de início:"
                dataInicio = celula.Offset(0, 1).Value

            Case "Previsão de término:"
                dataFinal = celula.Offset(0, 1).Value

            Case "Desconto para pagamento antecipado:"
                ' Pode vir como 0,05 (5%) ou 5 (5%) dependendo do formato
                descontoValor = celula.Offset(0, 1).Value

        End Select

    Next celula

    If Len(nomeForma) = 0 Then
        MsgBox "Selecione a forma de pagamento na Tabela86.", vbExclamation
        Exit Sub
    End If

    ' ============================================================
    ' 2) NORMALIZAR DESCONTO (GARANTIR FRAÇÃO)
    ' ============================================================
    ' Se vier 5 (representando 5%), converte para 0,05
    If descontoValor > 1 Then
        descontoValor = descontoValor / 100
    End If

    ' ============================================================
    ' 3) LOCALIZAR AS LINHAS DA TABELA70 (SUBTOTAL / IMPOSTOS / TOTAL)
    ' ============================================================
    Set celulaSubtotal = Nothing
    Set celulaImpostos = Nothing
    Set celulaTotalGeral = Nothing
    Set celulaDesconto = Nothing

    For Each linhaTotal In tbl70.ListRows

        nomeLinha = Trim(UCase(linhaTotal.Range.Cells(1, 1).Value))

        If nomeLinha Like "*SUBTOTAL*" Then
            Set celulaSubtotal = linhaTotal.Range.Cells(1, 2)
        ElseIf nomeLinha Like "*IMPOSTOS*" Then
            Set celulaImpostos = linhaTotal.Range.Cells(1, 2)
        ElseIf nomeLinha Like "*DESCONTO*%*" Then
            Set celulaDesconto = linhaTotal.Range.Cells(1, 2)
        ElseIf nomeLinha Like "*TOTAL GERAL*" Then
            Set celulaTotalGeral = linhaTotal.Range.Cells(1, 2)
        End If

    Next linhaTotal

    If celulaSubtotal Is Nothing Then
        MsgBox "Não foi possível localizar a linha 'SUBTOTAL' na Tabela70.", vbCritical
        Exit Sub
    End If

    If celulaImpostos Is Nothing Then
        MsgBox "Não foi possível localizar a linha 'IMPOSTOS' na Tabela70.", vbCritical
        Exit Sub
    End If

    If celulaTotalGeral Is Nothing Then
        MsgBox "Não foi possível localizar a linha 'TOTAL GERAL' na Tabela70.", vbCritical
        Exit Sub
    End If

    ' ============================================================
    ' 4) BUSCAR A FORMA NA TABELA "Parcelas" (CALCULO_PISCINA)
    ' ============================================================
    linhaForma = 0

    With tblParcelas
        For i = 1 To .ListRows.Count
            If Trim(.DataBodyRange(i, 1).Value) = nomeForma Then
                linhaForma = i
                Exit For
            End If
        Next i
    End With

    If linhaForma = 0 Then
        MsgBox "Forma de pagamento '" & nomeForma & "' não encontrada na tabela Parcelas.", vbCritical
        Exit Sub
    End If

    ' ============================================================
    ' 5) LER PARÂMETROS DA FORMA (inclui Imposto %)
    ' ============================================================
    With tblParcelas

        entradaPct = .DataBodyRange(linhaForma, .ListColumns("Entrada em %").Index).Value / 100
        numParcelas = .DataBodyRange(linhaForma, .ListColumns("Qtd Parcelas").Index).Value
        prazo = .DataBodyRange(linhaForma, .ListColumns("Prazo 1ª parcela dias").Index).Value
        intervalo = .DataBodyRange(linhaForma, .ListColumns("Intervalo dias").Index).Value
        tipoForma = Trim(.DataBodyRange(linhaForma, .ListColumns("Tipo").Index).Value)

        ' NOVO: Imposto % (ex.: 9,38 ou 0)
        impostoPct = .DataBodyRange(linhaForma, .ListColumns("Imposto %").Index).Value

    End With

    ' Segurança: se vier vazio / texto, força 0
    If Not IsNumeric(impostoPct) Then impostoPct = 0
    If impostoPct < 0 Then impostoPct = 0

    ' ============================================================
    ' 6) ATUALIZAR "DESCONTO %:" NA TABELA70 (sem alterar formatação)
    ' ============================================================
    ' Mantém seu comportamento: o desconto exibido na Tabela70 reflete a forma escolhida
    If Not celulaDesconto Is Nothing Then
        If nomeForma = "À vista antecipado" Or InStr(1, UCase(nomeForma), "DINHEIRO", vbTextCompare) > 0 Then
            celulaDesconto.Value = descontoValor
        Else
            celulaDesconto.Value = 0
        End If
    End If

    ' ============================================================
    ' 7) ATUALIZAR FÓRMULA DE IMPOSTOS NA TABELA70 COM BASE NO "Imposto %"
    ' ============================================================
    ' Não "apaga" a fórmula: regrava a fórmula correta conforme a forma.
    ' Fórmula gerada:
    '   = <celulaSubtotal> * (impostoPct) / 100
    '
    ' Exemplo:
    '   =G200*9,38/100
    '
    ' Obs.: Usa o endereço real da célula de SUBTOTAL da Tabela70 (robusto)
    celulaImpostos.Formula = "=" & celulaSubtotal.Address(False, False) & "*" & Replace(CStr(impostoPct), ",", ".") & "/100"

    ' Garante recálculo antes de ler TOTAL GERAL
    Application.Calculate

    ' ============================================================
    ' 8) VALOR TOTAL PARA PARCELAMENTO = TOTAL GERAL (já recalculado)
    ' ============================================================
    valorTotal = celulaTotalGeral.Value

    If valorTotal <= 0 Then
        MsgBox "Valor total inválido para parcelamento (TOTAL GERAL).", vbCritical
        Exit Sub
    End If

    ' ============================================================
    ' 9) DEFINIR LINHA BASE (abaixo da Tabela86) E LIMPAR ÁREA
    ' ============================================================
    linhaBase = tbl86.Range.Rows.Count + tbl86.Range.Row + 1

    With ws.Range("A" & linhaBase & ":D" & linhaBase + 40)
        .Clear
        .Interior.Color = RGB(255, 255, 255)
        .Borders.LineStyle = xlNone
    End With

    ' Cabeçalho
    With ws.Range("A" & linhaBase & ":D" & linhaBase)
        .Value = Array("Parcela", "Data", "Valor (R$)", "-")
        .Font.Bold = True
        .Interior.Color = RGB(237, 125, 49)
    End With

    ' ============================================================
    ' 10) CÁLCULO DAS PARCELAS (mantido) + controle de soma
    ' ============================================================
    valorEntrada = Round(valorTotal * entradaPct, 2)

    If numParcelas > 1 Then
        valorParcela = Round((valorTotal - valorEntrada) / (numParcelas - 1), 2)
    Else
        valorParcela = 0
    End If

    dataBase = Date + prazo

    somaParcelas = 0

    ' ============================================================
    ' 11) GERAR AS LINHAS DAS PARCELAS (mantido)
    ' ============================================================
    For i = 1 To numParcelas

        ws.Cells(linhaBase + i, 1).Value = i

        Select Case tipoForma

            Case "Entrada/Final"
                If i = 1 Then
                    dataParcela = dataBase
                    ws.Cells(linhaBase + i, 3).Value = valorEntrada
                Else
                    dataParcela = dataFinal
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                End If

            Case "Entrada/Inico/Final"
                If i = 1 Then
                    dataParcela = dataBase
                    ws.Cells(linhaBase + i, 3).Value = valorEntrada
                ElseIf i = 2 Then
                    dataParcela = dataInicio
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                Else
                    dataParcela = dataFinal
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                End If

            Case "Entrada + Quinzenas"
                If i = 1 Then
                    dataParcela = dataBase
                    ws.Cells(linhaBase + i, 3).Value = valorEntrada
                Else
                    If IsNumeric(intervalo) Then
                        dataParcela = dataBase + ((i - 1) * CLng(intervalo))
                    Else
                        dataParcela = dataBase
                    End If
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                End If

            Case Else
                If i = 1 Then
                    dataParcela = dataBase
                    ws.Cells(linhaBase + i, 3).Value = valorEntrada
                ElseIf numParcelas = 2 And InStr(1, nomeForma, "final", vbTextCompare) > 0 And i = 2 Then
                    dataParcela = dataFinal
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                Else
                    If IsNumeric(intervalo) Then
                        dataParcela = dataBase + ((i - 1) * CLng(intervalo))
                    Else
                        dataParcela = dataBase
                    End If
                    ws.Cells(linhaBase + i, 3).Value = valorParcela
                End If

        End Select

        ws.Cells(linhaBase + i, 2).Value = dataParcela
        ws.Cells(linhaBase + i, 2).NumberFormat = "dd/mm/yyyy"
        ws.Cells(linhaBase + i, 3).NumberFormat = "$ #,##0.00"

        somaParcelas = somaParcelas + CDbl(ws.Cells(linhaBase + i, 3).Value)

        ' Bordas padrão (mantido)
        With ws.Range(ws.Cells(linhaBase + i, 1), ws.Cells(linhaBase + i, 3))
            .Interior.Color = RGB(255, 255, 255)
            .Borders(xlEdgeLeft).LineStyle = xlContinuous
            .Borders(xlEdgeTop).LineStyle = xlContinuous
            .Borders(xlEdgeBottom).LineStyle = xlContinuous
            .Borders(xlEdgeRight).LineStyle = xlContinuous
            .Borders.Color = RGB(237, 125, 49)
        End With

    Next i

    ' ============================================================
    ' 12) AJUSTE FINAL DE CENTAVOS (soma das parcelas = valorTotal)
    ' ============================================================
    diferenca = Round(valorTotal - somaParcelas, 2)

    If Abs(diferenca) >= 0.01 Then
        ws.Cells(linhaBase + numParcelas, 3).Value = Round(ws.Cells(linhaBase + numParcelas, 3).Value + diferenca, 2)
        ws.Cells(linhaBase + numParcelas, 3).NumberFormat = "$ #,##0.00"
    End If

    ' ============================================================
    ' 13) CRIAR A LISTOBJECT (TabelaParcelas) E PROTEGER (mantido)
    ' ============================================================
    ws.Unprotect Password:="*Ju290480"

    Set rngDestino = ws.Range("A" & linhaBase).CurrentRegion
    Set tblCriada = ws.ListObjects.Add(xlSrcRange, rngDestino, , xlYes)

    tblCriada.Name = "TabelaParcelas"
    tblCriada.TableStyle = "TableStyleLight1"

    ws.Protect Password:="*Ju290480", _
        DrawingObjects:=True, _
        Contents:=True, _
        Scenarios:=True, _
        AllowUsingPivotTables:=True

End Sub

VBA MACRO Modulo_FecharNegocio.bas 
in file: xl/vbaProject.bin - OLE stream: 'VBA/Modulo_FecharNegocio'

Option Explicit

' =====================================================================
' Modulo_FecharNegocio (v4)
' - Read manual plano or auto-split
' - Adjust Linear!Tabela70 DESCONTO% so TOTAL GERAL = Valor Acordado
' - RefreshAll so Imp.Tabela picks up new values
' =====================================================================

Public Sub FecharNegocio()

    Dim ws As Worksheet
    Set ws = ThisWorkbook.Sheets("Linear")

    Dim valorAcordado As Double
    Dim qtdParcelas As Long
    Dim dataPrimeira As Date
    Dim intervalo As Long

    Dim i As Long
    Dim linhaPlano As Long
    Dim linhaBase As Long
    Dim planeadoLinhaInicio As Long
    Dim vData As Variant, vValor As Variant
    Dim temData As Boolean, temValor As Boolean
    Dim totalManual As Double
    Dim qtdManual As Long
    Dim valorParcela As Double
    Dim dataParcela As Date
    Dim somaParcelas As Double
    Dim diferenca As Double
    Dim resposta As VbMsgBoxResult

    On Error Resume Next
    valorAcordado = CDbl(ws.Range("FN_VALOR_ACORDADO").Value)
    qtdParcelas = CLng(ws.Range("FN_QTD_PARCELAS").Value)
    dataPrimeira = CDate(ws.Range("FN_DATA_PRIMEIRA").Value)
    intervalo = CLng(ws.Range("FN_INTERVALO").Value)
    On Error GoTo 0

    If valorAcordado <= 0 Then
        MsgBox "Informe o VALOR ACORDADO.", vbExclamation
        Exit Sub
    End If
    If qtdParcelas <= 0 Then qtdParcelas = 1
    If intervalo <= 0 Then intervalo = 30
    If dataPrimeira < Date Then dataPrimeira = Date + 1

    planeadoLinhaInicio = 294

    ' -----------------------------------------------------------------
    ' Ler plano manual (opcional)
    ' -----------------------------------------------------------------
    totalManual = 0
    qtdManual = 0
    For i = 1 To qtdParcelas
        linhaPlano = planeadoLinhaInicio + (i - 1)
        vData = ws.Cells(linhaPlano, 2).Value
        vValor = ws.Cells(linhaPlano, 3).Value
        temData = (IsDate(vData) Or (IsNumeric(vData) And CDbl(vData) > 0))
        temValor = (IsNumeric(vValor) And CDbl(vValor) > 0)
        If temData And temValor Then
            qtdManual = qtdManual + 1
            totalManual = totalManual + CDbl(vValor)
        End If
    Next i

    If qtdManual > 0 And qtdManual < qtdParcelas Then
        MsgBox "Voce preencheu " & qtdManual & " de " & qtdParcelas & _
               " parcelas manualmente. Preencha todas ou deixe todas em branco para divisao automatica.", _
               vbExclamation
        Exit Sub
    End If

    If qtdManual = qtdParcelas Then
        diferenca = Round(valorAcordado - totalManual, 2)
        If Abs(diferenca) >= 0.01 Then
            resposta = MsgBox( _
                "A soma das parcelas (R$ " & Format(totalManual, "#,##0.00") & ") e diferente do VALOR ACORDADO (R$ " & _
                Format(valorAcordado, "#,##0.00") & ")." & vbCrLf & vbCrLf & _
                "Diferenca: R$ " & Format(diferenca, "#,##0.00") & vbCrLf & vbCrLf & _
                "Deseja continuar assim mesmo?", _
                vbYesNo + vbQuestion, "Soma nao confere")
            If resposta = vbNo Then Exit Sub
        End If
    End If

    ' -----------------------------------------------------------------
    ' Desabilita eventos / tela / desprotege TODAS as planilhas relevantes
    ' antes de fazer qualquer modificacao
    ' -----------------------------------------------------------------
    Application.EnableEvents = False
    Application.ScreenUpdating = False

    Dim wsImp As Worksheet
    On Error Resume Next
    Set wsImp = ThisWorkbook.Sheets("Imp.Tabela")
    On Error GoTo 0

    ws.Unprotect Password:="*Ju290480"
    On Error Resume Next
    wsImp.Unprotect Password:="*Ju290480"
    On Error GoTo 0

    ' -----------------------------------------------------------------
    ' AJUSTE TABELA70: back-calcular DESCONTO % para que TOTAL GERAL = Valor Acordado
    ' TOTAL = SUBTOTAL + IMPOSTOS - (SUBTOTAL * DESCONTO%)
    ' => DESCONTO% = (SUBTOTAL + IMPOSTOS - VALOR_ACORDADO) / SUBTOTAL
    ' -----------------------------------------------------------------
    Dim subtotal As Double, impostos As Double, descontoNovo As Double
    subtotal = CDbl(ws.Range("G200").Value)
    impostos = CDbl(ws.Range("G201").Value)

    If subtotal > 0 Then
        descontoNovo = (subtotal + impostos - valorAcordado) / subtotal
        ' Escreve valor direto (substitui a formula 0,05 original)
        ws.Range("G202").Value = descontoNovo
    End If

    ' -----------------------------------------------------------------
    ' REGRAVA TABELAPARCELAS
    ' -----------------------------------------------------------------
    Dim lo As ListObject
    On Error Resume Next
    For Each lo In ws.ListObjects
        If lo.Name = "TabelaParcelas" Then
            lo.Unlist
            Exit For
        End If
    Next lo
    On Error GoTo 0

    linhaBase = 214
    With ws.Range("A" & linhaBase & ":D" & (linhaBase + 22))
        .Clear
        .Interior.Color = RGB(255, 255, 255)
        .Borders.LineStyle = xlNone
    End With

    With ws.Range("A" & linhaBase & ":D" & linhaBase)
        .Value = Array("Parcela", "Data", "Valor (R$)", "-")
        .Font.Bold = True
        .Interior.Color = RGB(237, 125, 49)
        .Font.Color = RGB(255, 255, 255)
        .HorizontalAlignment = xlCenter
    End With

    somaParcelas = 0

    For i = 1 To qtdParcelas
        If qtdManual = qtdParcelas Then
            linhaPlano = planeadoLinhaInicio + (i - 1)
            dataParcela = CDate(ws.Cells(linhaPlano, 2).Value)
            valorParcela = CDbl(ws.Cells(linhaPlano, 3).Value)
        Else
            dataParcela = dataPrimeira + ((i - 1) * intervalo)
            valorParcela = Int(valorAcordado / qtdParcelas * 100) / 100
        End If

        ws.Cells(linhaBase + i, 1).Value = i
        ws.Cells(linhaBase + i, 2).Value = dataParcela
        ws.Cells(linhaBase + i, 2).NumberFormat = "dd/mm/yyyy"
        ws.Cells(linhaBase + i, 3).Value = valorParcela
        ws.Cells(linhaBase + i, 3).NumberFormat = "$ #,##0.00"

        somaParcelas = somaParcelas + valorParcela

        With ws.Range(ws.Cells(linhaBase + i, 1), ws.Cells(linhaBase + i, 3))
            .Borders(xlEdgeLeft).LineStyle = xlContinuous
            .Borders(xlEdgeTop).LineStyle = xlContinuous
            .Borders(xlEdgeBottom).LineStyle = xlContinuous
            .Borders(xlEdgeRight).LineStyle = xlContinuous
            .Borders.Color = RGB(237, 125, 49)
        End With
    Next i

    If qtdManual <> qtdParcelas Then
        diferenca = Round(valorAcordado - somaParcelas, 2)
        If Abs(diferenca) >= 0.01 Then
            ws.Cells(linhaBase + qtdParcelas, 3).Value = _
                Round(ws.Cells(linhaBase + qtdParcelas, 3).Value + diferenca, 2)
        End If
    End If

    ' IMPORTANTE: ListObject precisa das 4 colunas (A:D) — Power Query "TabelaParcelas"
    ' espera cabecalho "Parcela | Data | Valor (R$) | -". A coluna D pode ficar vazia.
    Dim rngDestino As Range
    Dim tblNovo As ListObject
    Set rngDestino = ws.Range("A" & linhaBase & ":D" & (linhaBase + qtdParcelas))
    Set tblNovo = ws.ListObjects.Add(xlSrcRange, rngDestino, , xlYes)
    tblNovo.Name = "TabelaParcelas"
    tblNovo.TableStyle = "TableStyleLight1"

    ' -----------------------------------------------------------------
    ' REFRESH: calcula e atualiza queries. Linear e Imp.Tabela estao
    ' desprotegidas desde o inicio, entao writes sao seguros.
    ' -----------------------------------------------------------------
    Application.Calculate
    DoEvents

    On Error Resume Next
    ThisWorkbook.RefreshAll
    On Error GoTo 0
    DoEvents

    ' Da tempo das queries assincronas terminarem
    Application.Wait Now + TimeValue("0:00:02")
    DoEvents

    ' Atualiza pivot tables se existirem
    Dim pt As PivotTable
    On Error Resume Next
    For Each pt In wsImp.PivotTables
        pt.RefreshTable
    Next pt
    On Error GoTo 0
    DoEvents

    Application.ScreenUpdating = True
    Application.EnableEvents = True

    ' Reprotege 7 segundos depois, dando tempo para queries assincronas
    ' completarem depois que o usuario fechar o MsgBox.
    ' (Mesmo padrao do VBA original AtualizarImpTabelaCompleto.)
    Application.OnTime Now + TimeValue("00:00:07"), "ReproteherAposFecharNegocio"

    Dim modo As String
    If qtdManual = qtdParcelas Then
        modo = "valores MANUAIS do Plano de Parcelamento"
    Else
        modo = "divisao automatica (valores iguais)"
    End If

    MsgBox "Negocio fechado com " & modo & "!" & vbCrLf & vbCrLf & _
           "Valor total: R$ " & Format(valorAcordado, "#,##0.00") & vbCrLf & _
           qtdParcelas & " parcelas, soma final: R$ " & Format(somaParcelas, "#,##0.00") & vbCrLf & vbCrLf & _
           "Desconto ajustado: " & Format(descontoNovo, "0.00%") & vbCrLf & _
           "Agora o TOTAL GERAL (Linear G203 e Imp.Tabela) = Valor Acordado.", _
           vbInformation

End Sub
